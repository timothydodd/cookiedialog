## Why CookieDialog?

CookieDialog provides a simple, lightweight solution for GDPR cookie consent management that respects your users' privacy while keeping your site compliant.

<CardGrid stagger>
  <Card title="Zero Dependencies" icon="rocket">
    Pure JavaScript implementation with no external dependencies. Less than 20KB gzipped.
  </Card>
  <Card title="Geolocation Detection" icon="star">
    Automatically detect if users are in GDPR-required regions and show consent dialog accordingly.
  </Card>
  <Card title="Easy Integration" icon="add-document">
    Simple CDN or NPM installation. Works with any framework or vanilla JavaScript.
  </Card>
  <Card title="Fully Customizable" icon="pencil">
    Themes, translations, categories, and styling options to match your brand.
  </Card>
</CardGrid>

## Quick Installation

### CDN

```html
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/cookiedialog@1.0.3/dist/cookiedialog.min.css">
<script src="https://cdn.jsdelivr.net/npm/cookiedialog@1.0.3/dist/cookiedialog.min.js"></script>
```

### NPM

```bash
npm install cookiedialog
```

## Basic Usage

```javascript
CookieDialog.init({
  enableLocation: false, // false: Always show dialog, true: Only show in GDPR regions (requires geolocation)
  privacyUrl: '/privacy',
  onAccept: (consent) => {
    console.log('Cookies accepted:', consent);
  }
});
```

document.addEventListener('DOMContentLoaded', function() {
  let demoDialog = null;

  function showDemoDialog() {
    if (!demoDialog) {
      demoDialog = window.CookieDialog.init({
        forceShow: true,
        position: 'center',
        theme: 'light',
        categories: [
          { id: 'necessary', name: 'Essential', required: true },
          { id: 'analytics', name: 'Analytics', required: false },
          { id: 'marketing', name: 'Marketing', required: false }
        ],
        texts: {
          title: 'ðŸª CookieDialog Demo',
          description: 'This is a demo of the CookieDialog library. Choose your cookie preferences below.'
        },
        onAccept: (consent) => {
          console.log('Demo: Accepted cookies', consent);
        },
        onReject: (consent) => {
          console.log('Demo: Rejected cookies', consent);
        }
      });
    }
    demoDialog.show();
  }

  function resetDemoDialog() {
    if (demoDialog) {
      demoDialog.resetConsent();
      demoDialog.destroy();
      demoDialog = null;
    }
    localStorage.removeItem('cookiedialog_consent');
    console.log('Demo reset');
  }

  // Add event listeners
  const showBtn = document.getElementById('show-demo-btn');
  const resetBtn = document.getElementById('reset-demo-btn');
  
  if (showBtn) {
    showBtn.addEventListener('click', showDemoDialog);
  }
  
  if (resetBtn) {
    resetBtn.addEventListener('click', resetDemoDialog);
  }
});
`}
</script>
